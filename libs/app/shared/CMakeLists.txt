cmake_minimum_required( VERSION ${MIN_REQ_CMAKE_VERSION} )
include( ${CMAKE_CURRENT_LIST_DIR}/../../../build/functions.cmake )
project( Jde.App.Shared VERSION ${JDE_VERSION} LANGUAGES CXX )
include( ${jdeRoot}/build/dependencies.cmake )

set( targetName Jde.App.Shared )
add_library( ${targetName} )
compileOptions( ${targetName} )
target_link_libraries( ${targetName} Boost::json fmt::fmt )
add_dependencies( ${targetName} Jde.Web.Client ) #PCH includes Web.FromServer.pb.h, generated by Jde.Web.Client's protoc/linkGeneratedHeader steps.
target_include_directories( ${targetName} PUBLIC ${jdeRoot}/include/jde/web/client/proto ) #PUBLIC: the PCH below is PUBLIC and includes headers from this dir.

file(GLOB sources *.cpp proto/*.cpp log/*.cpp )
file(GLOB headers  */*.h ${jdeRoot}/include/jde/app/shared/*.h )
file(GLOB protos proto/*.proto )
target_sources( ${targetName} PRIVATE ${sources} ${headers} ${protos} )

if( WIN32 )
	find_package( OpenSSL REQUIRED )
	target_link_libraries( ${targetName} OpenSSL::SSL )
	target_link_libraries( ${targetName} absl::cord absl::log_internal_check_op absl::log_internal_conditions absl::log_internal_message absl::raw_hash_set absl::status absl::statusor )
	target_link_libraries( ${targetName} ${CMAKE_INSTALL_PREFIX}/protobuf/lib/protobuf$<IF:$<CONFIG:Debug>,d,>.lib )
endif()

#######################################################
target_precompile_headers(${targetName}
  PUBLIC
	<jde/web/client/exports.h>
	<jde/web/client/proto/Web.FromServer.pb.h>
	<jde/fwk.h>
	PRIVATE
	pc.h
)

set( protoOutDir ${PROJECT_SOURCE_DIR}/proto )
protobuf_generate(
	TARGET ${targetName}
	IMPORT_DIRS proto ../../web/client/proto
	PROTOC_OUT_DIR ${protoOutDir}
)
suppressProtoWarnings( "${protos}" ${protoOutDir} )

linkGeneratedHeader( ${targetName} "${protoOutDir}/App.FromServer.pb.h" "${jdeRoot}/include/jde/app/proto/App.FromServer.pb.h" )
linkGeneratedHeader( ${targetName} "${protoOutDir}/App.FromClient.pb.h" "${jdeRoot}/include/jde/app/proto/App.FromClient.pb.h" )
linkGeneratedHeader( ${targetName} "${protoOutDir}/Common.pb.h" "${jdeRoot}/include/jde/app/proto/Common.pb.h" )
linkGeneratedHeader( ${targetName} "${protoOutDir}/App.pb.h" "${jdeRoot}/include/jde/app/proto/App.pb.h" )
linkGeneratedHeader( ${targetName} "${protoOutDir}/Log.pb.h" "${jdeRoot}/include/jde/app/proto/Log.pb.h" )
